>> Membuat pseudo code cara mengoperasikan Microsoft Word <<


<PROGRAM JUDUL= mengoperasikan Microsoft Word>

SET laptop = 1;
SET microsoft-office = 1;
fungsi #install = installing-app;
fungsi #sld = siapkan laptop dulu;
fungsi #hidupkan = nyala;
fungsi #mulai = membuka aplikasi;

ALGORITMA=
menyiapkan_laptop()
JIKA laptop < 1 MAKA
        tampilkan #sld();
    LAINNYA 
        tampilkan #hidupkan();
        AKHIRIJIKA;

KETIKA laptop == #hidupkan;
        #mulai microsoft-office;
    AKHIRIKETIKA

JIKA microsoft-office < 1 MAKA
        tampilkan #install microsoft-office();
    LAINNYA
        tampilkan #mulai microsoft-office
        AKHIRIJIKA

</PROGRAM>
